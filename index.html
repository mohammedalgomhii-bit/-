import os import re import requests from flask import Flask, request, jsonify app = Flask(__name__) # الرابط الخاص بك في SheetDB SHEETDB_API_URL = "https://sheetdb.io/api/v1/8ku63b4j58gja" # ذاكرة مؤقتة لحفظ حالة كل عميل أثناء المحادثة user_context = {} def validate_name(name): # التحقق من وجود اسم ثنائي أو كلمة تدل على مؤسسة org_keywords = ["شركة", "مستشفى", "مدرسة", "جمعية", "مؤسسة", "مركز", "فندق"] is_org = any(key in name for key in org_keywords) is_double_name = len(name.strip().split()) >= 2 return is_org or is_double_name def validate_phone(phone): # التحقق من الرقم: 9 أرقام ويبدأ بـ 77 أو 78 أو 73 return bool(re.match(r"^(77|78|73)\d{7}$", phone)) @app.route("/", methods=["POST"]) @app.route("/webhook", methods=["POST"]) def chatbot(): data = request.json sender = data.get("sender") # معرف العميل (رقم هاتفه أو ID) message = data.get("message", "").strip() if sender not in user_context: user_context[sender] = {"step": "welcome"} state = user_context[sender] # الترويسة البصرية (ألوان بيضاء مع لمسات برتقالية وزرقاء عبر الرموز) header = "┏━━━━━━━━━━━━━━━┓\n" \ " *مطاعم الكورنيش السياحية* 🟠\n" \ " الرد الآلي لخدمة العملاء VIP 🔵\n" \ "┗━━━━━━━━━━━━━━━┛\n\n" # 1. مرحلة الترحيب وطلب الاسم if state["step"] == "welcome": response = header + "أهلاً بك في خدمة العملاء VIP. يرجى التكرم بكتابة (الاسم الثنائي) أو (اسم المنشأة/الجهة):" state["step"] = "get_name" return jsonify({"reply": response}) # 2. التحقق من الاسم elif state["step"] == "get_name": if validate_name(message): state["name"] = message response = "شكراً لك أستاذ/ة " + message + ".\nيرجى تزويدنا برقم هاتفك (9 أرقام يبدأ بـ 77، 78، أو 73):" state["step"] = "get_phone" else: response = "عذراً، يجب إدخال اسم ثنائي أو اسم الجهة (شركة/مستشفى/إلخ) لنتمكن من متابعة طلبك:" return jsonify({"reply": response}) # 3. التحقق من رقم الهاتف elif state["step"] == "get_phone": if validate_phone(message): state["phone"] = message response = "كيف يمكننا خدمتك اليوم؟ يرجى اختيار نوع الطلب:\n\n1️⃣ مناسبة\n2️⃣ طلب عقد تغذية\n3️⃣ طلب شخصي" state["step"] = "get_type" else: response = "الرقم غير صحيح. يرجى إدخال رقم يبدأ بـ (77 - 78 - 73) ومكون من 9 أرقام:" return jsonify({"reply": response}) # 4. توجيه الطلب (شخصي أو مناسبة) elif state["step"] == "get_type": if "شخصي" in message or "3" in message: response = "للطلبات الشخصية، يسعدنا خدمتك عبر تطبيق (لبيه) مباشرة:\nhttps://play.google.com/store/apps/details?id=com.techsoft.daily_dish_app" user_context.pop(sender) # إنهاء الجلسة elif "مناسبة" in message or "عقد" in message or "1" in message or "2" in message: state["type"] = message response = "يرجى كتابة تفاصيل الطلب حسب التنسيق التالي:\n\nالوجبات:\nالمشروبات:\nالسلطات:\nالتحلية:\nملاحظات:" state["step"] = "get_details" else: response = "يرجى اختيار أحد الخيارات (مناسبة / عقد تغذية / طلب شخصي)." return jsonify({"reply": response}) # 5. استلام التفاصيل وطلب التاريخ elif state["step"] == "get_details": state["details"] = message response = "متى موعد المناسبة؟ يرجى كتابة التاريخ بهذا التنسيق (مثال: 10/7/2026):" state["step"] = "get_date" return jsonify({"reply": response}) # 6. حفظ البيانات النهائية في قوقل شيت عبر SheetDB elif state["step"] == "get_date": state["date"] = message # تجهيز البيانات لتطابق أعمدة الشيت الخاص بك payload = { "data": [{ "name": state["name"], "phone": state["phone"], "date": state["date"], "type": state["type"], "details": state["details"], "timestamp": "Current Time" }] } try: res = requests.post(SHEETDB_API_URL, json=payload) if res.status_code == 201: response = "✅ تم تسجيل طلبك بنجاح في مطاعم الكورنيش السياحية. سيقوم القسم المختص بالتواصل معك قريباً." else: response = "عذراً، هناك مشكلة في الاتصال بالسيرفر. يرجى المحاولة لاحقاً." except: response = "حدث خطأ أثناء الحفظ. يرجى التواصل معنا هاتفياً." user_context.pop(sender) # مسح الذاكرة بعد النجاح return jsonify({"reply": response}) if __name__ == "__main__": app.run(debug=True)
